import flet as ft
import sqlite3
from datetime import datetime

def inicializar_banco():
    conn = sqlite3.connect("meu_caixa.db", check_same_thread=False)
    cursor = conn.cursor()
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS lancamentos (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            tipo TEXT,
            valor REAL,
            descricao TEXT,
            data TEXT
        )
    ''')
    conn.commit()
    return conn

def main(page: ft.Page):
    page.title = "Caixa - Posto Janjão"
    page.theme_mode = ft.ThemeMode.DARK
    page.window_width = 400
    page.vertical_alignment = ft.MainAxisAlignment.START
    page.scroll = ft.ScrollMode.AUTO

    conn = inicializar_banco()

    def obter_totais():
        cursor = conn.cursor()
        cursor.execute("SELECT tipo, SUM(valor) FROM lancamentos GROUP BY tipo")
        totais = {tipo: valor for tipo, valor in cursor.fetchall()}
        
        dinheiro = totais.get("Dinheiro", 0.0)
        pix = totais.get("Pix", 0.0)
        sangria = totais.get("Sangria", 0.0)
        
        lista_cartoes = [
            "Master Crédito", "Master Débito", 
            "Visa Crédito", "Visa Débito", 
            "Elo Crédito", "Elo Débito"
        ]
        total_cartoes = sum(totais.get(c, 0.0) for c in lista_cartoes)
        
        fisico = dinheiro - sangria
        return fisico, pix, total_cartoes

    txt_fisico = ft.Text("R$ 0.00", size=45, weight=ft.FontWeight.BOLD, color=ft.Colors.GREEN_400)
    txt_pix = ft.Text("R$ 0.00", size=22, weight=ft.FontWeight.BOLD, color=ft.Colors.BLUE_400)
    txt_cartoes = ft.Text("R$ 0.00", size=22, weight=ft.FontWeight.BOLD, color=ft.Colors.ORANGE_400)

    def atualizar_painel():
        fisico, pix, cartoes = obter_totais()
        txt_fisico.value = f"R$ {fisico:.2f}"
        txt_pix.value = f"R$ {pix:.2f}"
        txt_cartoes.value = f"R$ {cartoes:.2f}"
        page.update()

    dropdown_tipo = ft.Dropdown(
        label="Forma de Pagamento",
        options=[
            ft.dropdown.Option("Dinheiro"),
            ft.dropdown.Option("Pix"),
            ft.dropdown.Option("Master Crédito"),
            ft.dropdown.Option("Master Débito"),
            ft.dropdown.Option("Visa Crédito"),
            ft.dropdown.Option("Visa Débito"),
            ft.dropdown.Option("Elo Crédito"),
            ft.dropdown.Option("Elo Débito"),
            ft.dropdown.Option("Sangria"),
        ],
        value="Dinheiro",
        width=300
    )

    input_valor = ft.TextField(
        label="Valor (Ex: 50.00 ou 50,00)", 
        width=300,
        prefix=ft.Text("R$ ")
    )
    
    input_desc = ft.TextField(
        label="Descrição / Placa (Opcional)",
        width=300
    )

    lista_agrupada = ft.ListView(expand=True, spacing=10, height=300)

    # --- LISTA REFEITA PARA AGRUPAR E SOMAR AS BANDEIRAS ---
    def carregar_lista_agrupada():
        lista_agrupada.controls.clear()
        cursor = conn.cursor()
        # O SEGREDO ESTÁ AQUI: SUM(valor) GROUP BY tipo
        cursor.execute("SELECT tipo, SUM(valor) FROM lancamentos GROUP BY tipo")
        
        for tipo, valor_total in cursor.fetchall():
            if valor_total == 0:
                continue
                
            if tipo == "Dinheiro":
                cor = ft.Colors.GREEN
                icone = ft.Icons.MONEY
            elif tipo == "Sangria":
                cor = ft.Colors.RED_400
                icone = ft.Icons.REMOVE_CIRCLE
            elif tipo == "Pix":
                cor = ft.Colors.BLUE_400
                icone = ft.Icons.PIX
            else: 
                cor = ft.Colors.ORANGE_400
                icone = ft.Icons.CREDIT_CARD
            
            lista_agrupada.controls.append(
                ft.ListTile(
                    leading=ft.Icon(icone, color=cor),
                    title=ft.Text(f"{tipo} - R$ {valor_total:.2f}", color=cor, weight=ft.FontWeight.BOLD)
                )
            )
        page.update()

    def acao_lancar(e):
        if not input_valor.value:
            return
        
        try:
            valor_float = float(input_valor.value.replace(",", "."))
        except ValueError:
            input_valor.error_text = "Valor inválido"
            page.update()
            return

        data_atual = datetime.now().strftime("%H:%M - %d/%m")
        
        cursor = conn.cursor()
        cursor.execute("INSERT INTO lancamentos (tipo, valor, descricao, data) VALUES (?, ?, ?, ?)",
                       (dropdown_tipo.value, valor_float, input_desc.value, data_atual))
        conn.commit()

        input_valor.value = ""
        input_desc.value = ""
        input_valor.error_text = None
        
        atualizar_painel()
        carregar_lista_agrupada()
        input_valor.focus()

    btn_lancar = ft.ElevatedButton(
        "Lançar Abastecimento", 
        icon=ft.Icons.ADD_SHOPPING_CART,
        on_click=acao_lancar,
        width=300,
        height=55,
        bgcolor=ft.Colors.BLUE_700,
        color=ft.Colors.WHITE
    )

    # --- BOTÃO ZERAR TUDO COM PROTEÇÃO ---
    dlg_limpar = ft.AlertDialog(
        modal=True,
        title=ft.Text("Zerar Tudo?"),
        content=ft.Text("Tem certeza que deseja apagar todos os lançamentos do turno de hoje?"),
    )

    def confirmar_limpeza(e):
        cursor = conn.cursor()
        cursor.execute("DELETE FROM lancamentos")
        conn.commit()
        page.close(dlg_limpar)
        atualizar_painel()
        carregar_lista_agrupada()
        page.update()

    def cancelar_limpeza(e):
        page.close(dlg_limpar)

    dlg_limpar.actions = [
        ft.TextButton("Cancelar", on_click=cancelar_limpeza),
        ft.ElevatedButton("Sim, Zerar", bgcolor=ft.Colors.RED_700, color=ft.Colors.WHITE, on_click=confirmar_limpeza)
    ]

    def acao_limpar_tudo(e):
        page.open(dlg_limpar)

    btn_limpar = ft.ElevatedButton(
        "Limpar / Zerar Tudo",
        icon=ft.Icons.DELETE,
        on_click=acao_limpar_tudo,
        width=300,
        height=45,
        bgcolor=ft.Colors.RED_700,
        color=ft.Colors.WHITE
    )

    dlg_modal = ft.AlertDialog(modal=True, title=ft.Text("Resumo do Turno"))

    def fechar_turno(e):
        cursor = conn.cursor()
        cursor.execute("DELETE FROM lancamentos")
        conn.commit()
        page.close(dlg_modal)
        atualizar_painel()
        carregar_lista_agrupada()

    def cancelar_fechamento(e):
        page.close(dlg_modal)

    def abrir_resumo(e):
        cursor = conn.cursor()
        cursor.execute("SELECT tipo, SUM(valor) FROM lancamentos GROUP BY tipo")
        totais = {tipo: valor for tipo, valor in cursor.fetchall()}
        
        dinheiro = totais.get("Dinheiro", 0.0)
        pix = totais.get("Pix", 0.0)
        sangria = totais.get("Sangria", 0.0)
        
        lista_cartoes = [
            "Master Crédito", "Master Débito", 
            "Visa Crédito", "Visa Débito", 
            "Elo Crédito", "Elo Débito"
        ]
        
        total_cartoes = sum(totais.get(cartao, 0.0) for cartao in lista_cartoes)
        
        fisico_esperado = dinheiro - sangria
        total_vendido = dinheiro + pix + total_cartoes
        
        resumo_conteudo = [
            ft.Text(f"💰 Dinheiro: R$ {dinheiro:.2f}"),
            ft.Text(f"📱 Pix: R$ {pix:.2f}"),
            ft.Divider(),
            ft.Text("💳 Detalhamento de Cartões:", weight=ft.FontWeight.BOLD, color=ft.Colors.ORANGE_400)
        ]

        tem_cartao = False
        for cartao in lista_cartoes:
            valor_cartao = totais.get(cartao, 0.0)
            if valor_cartao > 0:
                resumo_conteudo.append(ft.Text(f"   • {cartao}: R$ {valor_cartao:.2f}"))
                tem_cartao = True
        
        if not tem_cartao:
            resumo_conteudo.append(ft.Text("   Nenhum cartão passado.", color=ft.Colors.GREY_500))
        else:
            resumo_conteudo.append(
                ft.Text(f"   => TOTAL CARTÕES: R$ {total_cartoes:.2f}", weight=ft.FontWeight.BOLD, color=ft.Colors.ORANGE_300)
            )

        resumo_conteudo.extend([
            ft.Divider(),
            ft.Text(f"🩸 Sangrias (Retiradas): R$ {sangria:.2f}", color=ft.Colors.RED_400),
            ft.Divider(),
            ft.Text(f"💵 FÍSICO NA CARTEIRA: R$ {fisico_esperado:.2f}", weight=ft.FontWeight.BOLD, color=ft.Colors.GREEN_400),
            ft.Text(f"📊 TOTAL VENDIDO: R$ {total_vendido:.2f}", weight=ft.FontWeight.BOLD, color=ft.Colors.BLUE_400),
            ft.Text("\nDeseja finalizar este turno e zerar o caixa para amanhã?", size=12, color=ft.Colors.GREY_400)
        ])
        
        dlg_modal.content = ft.Column(resumo_conteudo, tight=True)
        
        dlg_modal.actions = [
            ft.TextButton("Cancelar", on_click=cancelar_fechamento),
            ft.ElevatedButton("Zerar Caixa", bgcolor=ft.Colors.RED_700, color=ft.Colors.WHITE, on_click=fechar_turno),
        ]
        
        page.open(dlg_modal)

    btn_resumo = ft.ElevatedButton(
        "Fechar Caixa / Resumo",
        icon=ft.Icons.ANALYTICS,
        on_click=abrir_resumo,
        width=300,
        height=45,
        bgcolor=ft.Colors.GREY_800,
        color=ft.Colors.WHITE
    )

    painel_somas = ft.Row([
        ft.Column([
            ft.Text("📱 Total PIX", color=ft.Colors.GREY_400, size=12),
            txt_pix
        ], horizontal_alignment=ft.CrossAxisAlignment.CENTER),
        ft.Container(width=40),
        ft.Column([
            ft.Text("💳 Total Cartões", color=ft.Colors.GREY_400, size=12),
            txt_cartoes
        ], horizontal_alignment=ft.CrossAxisAlignment.CENTER)
    ], alignment=ft.MainAxisAlignment.CENTER)

    page.add(
        ft.Column([
            ft.Text("Físico na Carteira (Dinheiro):", size=16, color=ft.Colors.GREY_400),
            txt_fisico,
            painel_somas,
            ft.Divider(height=20, color=ft.Colors.WHITE24),
            dropdown_tipo,
            input_valor,
            input_desc,
            btn_lancar,
            btn_resumo,
            btn_limpar, 
            ft.Divider(height=20, color=ft.Colors.WHITE24),
            ft.Text("Totais por Bandeira:", size=16, weight=ft.FontWeight.BOLD),
            lista_agrupada
        ], horizontal_alignment=ft.CrossAxisAlignment.CENTER)
    )

    atualizar_painel()
    carregar_lista_agrupada()

ft.app(target=main, view=ft.AppView.WEB_BROWSER, port=8000, host="0.0.0.0")
